21void motor_set_duty_cycle(Motor* motor, int32_t doot){
22 motor->duty_cycle = doot;
23 // First, check if the motor is disabled
24 if(motor->enable_flag != 1){
25 // if the enable flag isn't set exit the function and do nothing.
26 // we are also using != 1 so if there is a stray value in memory,
27 // the motor doesn't accidentally enable.
28 return;
29 }
30
31 // Next, saturate the duty cycle just in case.
32 if(doot < -100){
33 doot = -100;
34 }
35 if(doot > 100){
36 doot = 100;
37 }
38
39 // We need to get the auto reload value for the timer we are using
40 // signed value so we don't run into sign issues later
41 int32_t ARR = (int32_t)(motor->timer->Init.Period + 1);
42
43 // Now calculate the duty cycle in terms of the CCR value
44 doot = doot*ARR/100; // multiply first so we don't lose data
45
46 // now we need to set the motors to the correct duty cycles
47 // Forwards will be channels 1 and 3 for motors 1 and 2 respectively
48 // Backwards will be channels 2 and 4 for motors 1 and 2 respectively
49
50
51 // the below CCR's are based on the logic table of the toshiba, setting motor.
52 // to brake mode
53 // if duty cycle is <0
54 if (doot < 0){
55 // check if it is the first or second motor.
56 if(motor->channels == 1){
57 motor->timer->Instance->CCR1 = ARR;
58 motor->timer->Instance->CCR2 = ARR + doot;
59 } else if(motor->channels == 2){
60 motor->timer->Instance->CCR3 = ARR;
61 motor->timer->Instance->CCR4 = ARR + doot;
62 }else{
63 // if neither return
64 return;
65 }
66 // if duty cycle >=0
67 } else{
68 if(motor->channels == 1){
69 motor->timer->Instance->CCR1 = ARR - doot;
70 motor->timer->Instance->CCR2 = ARR;
71 } else if(motor->channels == 2){
72 motor->timer->Instance->CCR3 = ARR - doot;
73 motor->timer->Instance->CCR4 = ARR;
74 }else{
